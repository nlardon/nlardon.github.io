Activité Programmation

1 - Schéma 2 LEDs

schema2_led.ino

int led1 = 8;
int led2 = 4;
int bouton = 1;
int etatbouton = LOW;

void setup() {                
    pinMode(led1, OUTPUT);     
    pinMode(led2, OUTPUT);
    pinMode(bouton, INPUT_PULLUP);      
}

void loop() {

    etatbouton = digitalRead(bouton); //Si bouton appuyé retourne état BAS (LOW)

    if (etatbouton == LOW) {
        digitalWrite(led1, HIGH);
        digitalWrite(led2, LOW);
    }  
    else {
        digitalWrite(led1, LOW);
        digitalWrite(led2, HIGH);
    }
    delay(200);
}

Bandeau LED

Vérifier de bien avoir la bibliothéque : Adafruit_NeoPixel

#include <Adafruit_NeoPixel.h>

// Which pin on the Arduino is connected to the NeoPixels?
#define LED_PIN     5

// How many NeoPixels are attached to the Arduino?
#define LED_COUNT  5

// Declare our NeoPixel strip object:
Adafruit_NeoPixel strip(LED_COUNT, LED_PIN, NEO_GRBW + NEO_KHZ800);

int brightness = 50;

void setup() {
  strip.begin();           // INITIALIZE NeoPixel strip object (REQUIRED)
  strip.show();            // Turn OFF all pixels ASAP
  strip.setBrightness(brightness);
}

void loop() {

  for(int i=0; i<1; i++) {
      // pixels.Color takes RGBW values, from 0,0,0,0 up to 255,255,255,255
      strip.setPixelColor(i, strip.Color(  255,   0,   0, 0));         //  R G B W : Set pixel's color (in RAM)
      strip.show();                          //  Update strip to match
      delay(100);                           //  Pause for a moment
    }

  for(int i=1; i<4; i++) {
      strip.setPixelColor(i, strip.Color(  0,   255,   0, 0));         //  R G B W : Set pixel's color (in RAM)
      strip.show();                          //  Update strip to match
      delay(100);                           //  Pause for a moment
    }

for(int i=4; i<LED_COUNT; i++) {
      strip.setPixelColor(i, strip.Color(  0,   0,   255, 0));         //  R G B W : Set pixel's color (in RAM)
      strip.show();                          //  Update strip to match
      delay(100);                           //  Pause for a moment
    }

}

Anneau LED

Vérifier de bien avoir la bibliothéque : Adafruit_NeoPixel

#include <Adafruit_NeoPixel.h>
#define PIN 7 // input pin Neopixel is attached to
#define NUMPIXELS 16 // number of neopixels in Ring

Adafruit_NeoPixel pixels = Adafruit_NeoPixel(NUMPIXELS, PIN);

void setup() {
    pixels.begin(); // Initializes the NeoPixel library.
}

void loop() {

    for(int i=0;i<8;i++){
        // pixels.Color takes RGB values, from 0,0,0 up to 255,255,255
        pixels.setPixelColor(i, pixels.Color(0, 0, 255)); // R, G, B
        pixels.show(); // This sends the updated pixel color to the hardware.
        delay(100); // Delay for a period of time (in milliseconds).
    }

    for(int i=8;i<16;i++){
        pixels.setPixelColor(i, pixels.Color(255, 0, 0)); // R, G, B
        pixels.show(); // This sends the updated pixel color to the hardware.
        delay(100); // Delay for a period of time (in milliseconds).
    }

}
