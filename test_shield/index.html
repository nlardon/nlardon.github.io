void setup() {
  // put your setup code here, to run once:
  Serial.begin(9600);
  Serial.println("--- Test STEM Shield ---");

  pinMode(SW1, INPUT);
  pinMode(SW2, INPUT);
  pinMode(SW3, INPUT);
  pinMode(SW4, INPUT);
  pinMode(SW5, INPUT);
  pinMode(TRIMMER, INPUT);
  pinMode(NTC, INPUT);
  pinMode(PHOTO, INPUT);
}

void loop() {
  // put your main code here, to run repeatedly:

  // Test Trimmer
  int valuePot = analogRead(TRIMMER);
  Serial.print("Value Trimmer : ");
  Serial.println(valuePot);

  if (digitalRead(SW1) == HIGH) Serial.println("SW1 Switched ON !");
  if (digitalRead(SW2) == HIGH) Serial.println("SW2 Pressed !");
  if (digitalRead(SW3) == HIGH) Serial.println("SW3 Pressed !");
  if (digitalRead(SW4) == HIGH) Serial.println("SW4 Pressed !");
  if (digitalRead(SW5) == HIGH) Serial.println("SW5 Pressed !");

  Serial.print("Value NTC : ");
  Serial.println( analogRead(NTC)  ) ;

  Serial.print("Value Photo : ");
  Serial.println( analogRead(PHOTO)  ) ;

  delay(500);

}

const int SW1 = 8;
const int SW2 = 9;

const int TRIMMER = A0;
const int NTC = A1;
const int PHOTO = A2;

const int IRLED = 7;
const int RED1 = A3;
const int ORANGE1 = A4;
const int GREEN1 = 2;
const int RED2 = 3;
const int ORANGE2 = 4;
const int GREEN2 = 5;
const int BUZZER = 13;
const int SERVO = 6;

Servo myservo;  // create Servo object to control a servo

void setup() {
  // put your setup code here, to run once:
  Serial.begin(9600);
  Serial.println("--- Test  STEM Shield  ---");

  pinMode(TRIMMER, INPUT);
  myservo.attach(SERVO);  // attaches the servo on pin 6 to the Servo object

}

void loop() {
  // put your main code here, to run repeatedly:

  // Test Trimmer
  int valuePot = analogRead(TRIMMER);
  Serial.print("Value Trimmer : ");
  Serial.println(valuePot);

  int thisPitch = map(valuePot, 400, 1000, 120, 1500);

  // play the pitch:
  tone(BUZZER, thisPitch, 10);
  delay(1);  // delay in between reads for stability

  int val = map(valuePot, 0, 1023, 0, 180);     // scale it for use with the servo (value between 0 and 180)
  myservo.write(valuePot);                  // sets the servo position according to the scaled value
  delay(15);                           // waits for the servo to get there

}
