Contrôle bandeau LED SK6812 avec une carte ESP32

LED SK6812

Une LED SK6812 est une LED adressable individuellement, souvent utilisée dans les bandes LED et d'autres applications d'éclairage décoratif.
Elle est similaire à la populaire WS2812B, mais offre des options de couleurs supplémentaires, notamment le blanc.

Caractéristiques principales :
- Adressable individuellement : Chaque LED SK6812 contient son propre circuit intégré (CI), ce qui permet de contrôler la couleur et la luminosité de chaque LED indépendamment.
- Plusieurs options de couleurs : Outre les couleurs rouge, vert et bleu (RGB), la SK6812 offre une option de blanc (RGBW).
- Type de LED : Généralement de type SMD 5050, ce qui signifie qu'elle mesure 5,0 mm x 5,0 mm.
- Communication : Utilise un protocole de communication série unipolaire, ce qui signifie qu'un seul fil de données est nécessaire pour contrôler plusieurs LED en cascade.
- Tension de fonctionnement : Généralement 5V ou 12V DC, selon le modèle.
- Intensité variable : Chaque couleur (rouge, vert, bleu et blanc) peut être modulée en intensité sur 256 niveaux (8 bits par couleur), offrant ainsi une large gamme de couleurs et d'effets de gradation.

Utilisation avec une carte ESP32

Il faut ajouter une bibliothéque à l'IDE Arduino pour utiliser ces LEDs. On pourrait coder sans blibliotéque suplémentaire mais le code serait complexe et difficilement lisible.

Nous allons utliser le travail du développeur "gitpeut" sur github. Ce travail nous simplifira le controle des LEDs.

Ajout de la bibliothéque (library)

Téléchager le projet du développeur : https://github.com/gitpeut/SK6812-RGBW-ESP32

Ajouter la bibliotèque dans l'IDE Arduino.

Code pour tester

#include <sk.h> //source : gitpeut/SK6812-RGBW-ESP32

sk strip; //Create an object to manipulate leds SK6812-RGBW
int led_count = 5; //Define the number of leds
int led_pin = 4; //Set the data pin

void setup()
{
  strip.begin(led_pin, led_count); //Initialize the strip
  strip.clear(); //Set to off all leds
  delay(100);
}


void loop()
{
  strip.clear(); //Set to off all leds

  strip.color(0,255,0,0,0); //Switch on the red on first led
  strip.color(1,0,255,0,0); //Switch on the green on second led
  strip.color(2,0,0,255,0); //Switch on the blue on third led
  strip.color(3,0,0,0,255); //Switch on the white on forth led
  strip.color(4,255,255,255,255); //Switch on the RGBW on fifth led

  strip.show(); //send the order to the leds
  delay(1000);

}

#include <sk.h> //source : gitpeut/SK6812-RGBW-ESP32

sk strip; //Create an object to manipulate leds SK6812-RGBW
int led_count = 5; //Define the number of leds
int led_pin = 4; //Set the data pin

void setup()
{
  strip.begin(led_pin, led_count); //Initialize the strip
  strip.clear(); //Set to off all leds
  delay(100);
}


void loop()
{
  strip.clear(); //Set to off all leds

  strip.color(0,255,0,0,0); //Switch on the red on first led
  strip.show(); //send the order to the leds
  delay(1000);

  strip.color(0,255,0,0,0); //Switch on the red on first led
  strip.color(1,0,255,0,0);
  strip.color(2,0,0,255,0);
  strip.color(3,0,0,0,255);
  strip.color(4,255,255,255,255);
  strip.show(); //send the order to the leds
  delay(1000);


  //Swith on full RGBW (0-255) of all leds
  for (int i=0;i<led_count;i++){
    strip.color(i,255,255,255,255);
  }
  strip.show();
  delay(1000);

  //Swith on red of all leds
  for (int i=0;i<led_count;i++){
    strip.color(i,255,0,0,0);
  }
  strip.show();
  delay(1000);

}

Code avec if et boucle + 3 boutons

#include <sk.h> //source : gitpeut/SK6812-RGBW-ESP32

sk strip; //Create an object to manipulate leds SK6812-RGBW
int led_count = 120; //Define the number of leds
int led_pin = 4; //Set the data pin

const int button1 = 12; 
const int button2 = 14; 
const int button3 = 27; 

bool etat_button1; 
bool etat_button2; 
bool etat_button3; 

void setup() {

    strip.begin(led_pin, led_count); //Initialize the strip
    strip.clear(); //Set to off all leds

    pinMode(button1, INPUT_PULLUP);
    pinMode(button2, INPUT_PULLUP);
    pinMode(button3, INPUT_PULLUP);
    delay(100);

}

void loop() {

    etat_button1 = digitalRead(button1);
    etat_button2 = digitalRead(button2);
    etat_button3 = digitalRead(button3);

    if (etat _button1 == LOW) {
        for(int i = 0 ; i < 5 ; i++) {
        strip.color(i,255,0,0,0); 
        } 
    } 

    else if (etat _button2 == LOW) {
        for(int i = 0 ; i < 5 ; i++) { 
        strip.color(i,0,255,0,0); 
        }
    }
    else if (etat _button3 == LOW) {
        for(int i = 0 ; i < 5 ; i++) { 
        strip.color(i,0,0,255,0); 
        }
    }
    else {
        for(int i = 0 ; i < 5 ; i++) {
        strip.color(i,0,0,0,0); 
        }
    }

    strip.show(); //send the order to the leds

    delay(10);
}
